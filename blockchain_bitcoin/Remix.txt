* 예제 ( Greeter.sol )

pragma solidity ^0.4.18;

contract Greeter {
     function sayHello() public pure returns(string) {
         return "Hello";
     }
 }



-------------------------------------------------------------------------------------------------------------------------------------

* 예제 Greeter_2.sol

pragma solidity ^0.4.18;

contract Greeter_2 {
     string message = "Hello";
     
     function sayHello() public view returns(string) {
         return message;
     }
     
     function changeHello(string _message) public {
         message = _message;
     }
 }
